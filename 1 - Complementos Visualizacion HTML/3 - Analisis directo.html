def create_chart(column_name, df):
    fig, axes = plt.subplots(1, 3, figsize=(18, 6))

    axes[0].boxplot(df[column_name], vert=False, patch_artist=True, 
                    boxprops=dict(facecolor='royalblue'))
    axes[0].set_title("Diagrama de Bigotes")
    axes[0].set_xlabel(column_name)

    sns.violinplot(x=column_name, data=df, ax=axes[1], color='royalblue')
    axes[1].set_title("Violin Plot - Distribucin de Datos")
    axes[1].set_xlabel(column_name)
    
    axes[2].hist(df[column_name], bins=50, color='royalblue', edgecolor='black')
    axes[2].set_title("Histograma de Datos")
    axes[2].set_xlabel(column_name)
    axes[2].set_ylabel("Frecuencia")
    
    plt.tight_layout()
    plt.show()

def get_bigquery_data(sql):
    query_job = client.query(sql)
    return query_job.to_dataframe()
